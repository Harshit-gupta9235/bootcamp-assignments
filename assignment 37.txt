6. Write a program to reverse vector elements.

ans:

#include<iostream>
#include<vector>
using namespace std;
int main(){
    vector<int>a;
    cout<<"enter how many no u want to input\n";
    int n;
    cin>>n;
    for (int i = 0; i < n; i++)
    {int k;
    cout<<"enter element\n";
    cin>>k;
     a.push_back(k);
    }
    a.reserve(a.size());
    vector<int>b=a;
    for (int i = b.size()-1; i>=0; i--)
    {
      cout<<b.at(i);
    }
    return 0;
}
